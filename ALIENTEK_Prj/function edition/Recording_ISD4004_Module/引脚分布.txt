RAC：ISD4004/4003 RAC 输出口
INT：ISD4004/4003 INT 输出口 
SCLK：ISD4004/4003 SCLK输出口 
MISO：ISD4004/4003 MISO输出口 
MOSI：ISD4004/4003 MOSI输出口 
SS：ISD4004/4003 SS输出口

VCC3.3:电源3.3V 电源输出/输入口
VCC5.0:电源5.0V 输出/输入口
GND: 模块地线 

void ISD4004_Rec(u16 addr)
{
	ISD4004_CS = 1;	
	ISD4004_CS = 0;	//拉低片选
	
	ISD4004_PwrUp();
	delay_ms(50);
	ISD4004_PwrUp();
	delay_ms(100);
	

	ISD4004_CS = 1;	
	ISD4004_CS = 0;	//拉低片选
	
	ISD4004(addr);
	ISD4004(ISD4004_SetRecAddr);
	ISD4004(ISD4004_RecAtCrtAddr);
	
	ISD4004_CS = 1;	

}
